%                                                                         %
%                                                                         %
%        *****   Ricardo's Boilerplate Packages for LaTeX   *****         %
%                                                                         %
%                                                                         %
%   You should copy-paste this block of code and add it at the            %
%   beginning of your paper, right before the title and author.           %
%                                                                         %
%   Note that some of these have been added to the ACL template           %
%   but not necessarily for other conference templates.                   %
%                                                                         %
%                                                                         %
%                         *****   Tables   *****                          %
%                                                                         %
\usepackage{rotating}   % Allows for sideways tables                      %
\usepackage{tabularx}   % Nice tables                                     %
\usepackage{multirow}   % Allows cells to span multiple rows              %
\usepackage{multicol}   % Allows cells to span multiple columns           %
\usepackage{booktabs}   % Nicer-looking tables & more formatting options  %
\usepackage{array}      % Adds more options for column formatting         %
%                                                                         %
%                                                                         %
%                         *****   Images   *****                          %
%                                                                         %
\usepackage{graphicx}   % Better image management                         %
\usepackage{svg}        % Allows for svg files to be used                 %
%                                                                         %
%                                                                         %
%                         *****   Floats   *****                          %
%                                                                         %
\usepackage{caption}    % More options for caption customization          %
\usepackage{subcaption} % Allows for captions in subfigures               %
\usepackage{expex}      % Lets you have numbered examples                 %
%                                                                         %
%                                                                         %
%                     *****   Text Formatting   *****                     %
%                                                                         %
\usepackage{xcolor}     % Lets us set the font color                      %
\usepackage{soul}       % Does text highlighting                          %
\usepackage[most]{tcolorbox}                                              %
%                       % Nice-looking boxes useful for prompts           %
%                                                                         %
%                                                                         %
%                       *****   Typography   *****                        %
%                                                                         %
\usepackage{xurl}       % Fixes up urls and makes them look better        %
\usepackage{microtype}  % Helps with layout & spacing                     %
%                                                                         %
%                                                                         %
%                          *****   Math   *****                           %
%                                                                         %
\usepackage{amsfonts}   % Adds support for special symbols used in math   %
\usepackage{amsmath}    % Extra support for mathematical typesetting      %
%                                                                         %
%                                                                         %
